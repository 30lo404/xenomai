139 static inline size_t xnheap_external_overhead(size_t hsize, size_t psize)
140 {
141  size_t pages = (hsize + psize - 1) / psize;
142  return xnheap_align(sizeof(struct xnextent)
143  + pages * sizeof(struct xnpagemap), psize);
144 }
145 
146 static inline size_t xnheap_internal_overhead(size_t hsize, size_t psize)
147 {
148  /* o = (h - o) * m / p + e
149  o * p = (h - o) * m + e * p
150  o * (p + m) = h * m + e * p
151  o = (h * m + e *p) / (p + m)
152  */
153  return xnheap_align((sizeof(struct xnextent) * psize
154  + sizeof(struct xnpagemap) * hsize)
155  / (psize + sizeof(struct xnpagemap)), psize);
156 }

161 static inline size_t xnheap_rounded_size(size_t hsize, size_t psize)
162 {
163  /*
164  * Account for the minimum heap size (i.e. 2 * page size) plus
165  * overhead so that the actual heap space is large enough to
166  * match the requested size. Using a small page size for large
167  * single-block heaps might reserve a lot of useless page map
168  * memory, but this should never get pathological anyway,
169  * since we only consume 4 bytes per page.
170  */
171  if (hsize < 2 * psize)
172  hsize = 2 * psize;
173  hsize += xnheap_external_overhead(hsize, psize);
174  return xnheap_align(hsize, psize);
175 }
